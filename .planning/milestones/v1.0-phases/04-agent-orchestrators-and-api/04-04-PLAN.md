```typescript
import type { FastifyInstance } from 'fastify'
import { prisma } from '@tcg/db'

export async function externalCardRoutes(fastify: FastifyInstance) {
  // Auth check helper
  const getUserId = (request: any) => {
    const userId = request.headers['x-user-id'] as string
    if (!userId) return null
    return userId
  }

  // POST /external-cards — create
  fastify.post('/external-cards', async (request, reply) => { ... })

  // PATCH /external-cards/:id — update
  fastify.patch('/external-cards/:id', async (request, reply) => { ... })

  // DELETE /external-cards/:id — soft delete
  fastify.delete('/external-cards/:id', async (request, reply) => { ... })

  // GET /external-cards/cert/:certNumber — PSA cert lookup stub
  fastify.get('/external-cards/cert/:certNumber', async (request, reply) => { ... })
}
```

**POST /external-cards:**
- Validate body: `title` (string, required) and `estimatedValue` (number, required). `set`, `grade`, `certNumber` are optional.
- Return 422 with error envelope if validation fails.
- Create via Prisma: `prisma.externalCard.create({ data: { userId, name: body.title, estimatedValue: body.estimatedValue, setName: body.set ?? null, grade: body.grade ?? null, certNumber: body.certNumber ?? null, priceConfidence: 'NO_DATA' } })`.
- Return 201 with the created record.
- Does NOT auto-trigger analysis (per CONTEXT.md locked decision).

**PATCH /external-cards/:id:**
- Find card: `prisma.externalCard.findFirst({ where: { id: params.id, userId, deletedAt: null } })`. Return 404 if not found.
- Update only provided fields from body (title, set, grade, certNumber, estimatedValue).
- Return 200 with updated record.

**DELETE /external-cards/:id:**
- Find card first (same query). Return 404 if not found.
- Soft delete: `prisma.externalCard.update({ where: { id }, data: { deletedAt: new Date() } })`.
- Return 200 with `{ status: 'deleted' }`.

**GET /external-cards/cert/:certNumber:**
- Query: `prisma.externalCard.findFirst({ where: { certNumber: params.certNumber, userId, deletedAt: null }, select: { grade: true } })`.
- Return 200 with `{ grade: card?.grade ?? null }`.

All endpoints use the consistent error envelope: `{ error: { code, message, details? } }`.
